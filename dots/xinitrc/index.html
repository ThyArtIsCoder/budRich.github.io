This is part of my anti-DM suite. The stuff I want to happen just before i3 loads up, goes here. Having urxvt in daemon mode is really nice. And I use xsetroot and a solid “greyed” color from Xresources instead of a wallpaper. I have written a little piece about the strange keyboard layout Seba. There is also a short text on why i use sxhkd.

#!/bin/sh
# .xinitrc - curated by budRich

xrdb -merge $HOME/.Xresources
xrandr --output eDP-1 --auto --left-of HDMI-1 --primary &
setxkbmap -layout seba -option ctrl:nocaps -option lv3:ralt_switch
hsetroot -tile /home/bud/etc/bitmap-walls/patterns-png/line_diag3_2.png \
  -tint "$(xrdb -query | awk '/\*background/ {print $2}')"
urxvtd -q -o -f
dropbox start &

igi &

killall polybar
killall qbittorrent
killall mpd

compton -b > /dev/null 2>&1
sxhkd > /dev/null 2>&1 &

xbanish > /dev/null 2>&1 &
mpd > /dev/null 2>&1 &

exec i3
