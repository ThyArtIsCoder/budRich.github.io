Polybar is a great statusbar that works well with i3 and other window managers, this is my config file. It uses some custom scripts to display some information.

ss-17-11-22-pbarro.png

In the screenshot above you can see polybar at the top of the screen. I use i3fyra to achieve the layout. Container A(vivaldi/wg) and B(irssi) are tabbed and container C is horizontally split with three urxvt instances.

Github dots/polybar/budbar 
; polybar config, curated by budRich since 2016

; >> pbtitle
[module/pbtitle]
type = custom/ipc
hook-0 = pbtitle "focus"
hook-1 = pbtitle title
hook-2 = pbtitle "normal"
hook-3 = pbtitle "advanced"
hook-4 = pbtitle color
hook-5 = pbtitle taskbar
initial = 3

; >> pbws
[module/pbws]
type = custom/ipc
hook-0 = pbws
hook-1 = pbws switch
hook-2 = pbws next
hook-3 = pbws prev
initial = 1

; >> pbrectime
[module/pbrectime]
type = custom/ipc
hook-0 = pbrectime start
hook-1 = pbrectime update
hook-2 = echo ""

; >> pbmedia
[module/pbmedia]
type = custom/ipc
hook-0 = pbmedia start
hook-1 = pbmedia "update"
hook-2 = pbmedia "resume"
hook-3 = echo ""
initial = 1 

; >> pbmic
[module/pbmic]
type = custom/ipc
hook-0 = pbmic
hook-1 = pbmic "mute"
hook-2 = pbmic "up"
hook-3 = pbmic "down"
initial = 1

; >> mpd
[module/mpd]
type = internal/mpd
host = 127.0.0.1
port = 6601

format-online = <label-song> <icon-repeat> <icon-random>
format-offline = <label-offline>
format-playing = ${self.format-online}
format-paused = ${self.format-online}
format-stopped = ${self.label-offline}

format-paused-foreground = ${colors.blk}
format-playing-foreground = ${colors.wht}

label-offline = turn on some music, dude
label-online = turn on some music, dude

icon-repeat = 
icon-random = 
icon-random-margin-right = 0
toggle-on-foreground = ${colors.blk}

label-song-maxlen = 35
label-song-ellipsis = true

; >> pbvol
[module/pbvol]
type = custom/script
tail = true
label = %output%
exec = ~/src/polybar/pbvol --listen
click-right = exec i3run -i pavucontrol -e pavucontrol
click-left = ~/src/polybar/pbvol --togmute
scroll-up = ~/src/polybar/pbvol --up
scroll-down = ~/src/polybar/pbvol --down
